# Is Subsequence 

``` java

class Solution {
    /**
    s가 t의 substring인지 확인
    si와 ti의 인덱스가 s,t 길이 이내인 동안 
    si, ti 인덱스 비교 -> 같으면 si++, ti++ 다르면 ti++
    se=s.length()-1
    종료 조건: si==se && s.charAt[se]==t.charAt[ti] 이면 return true
    return false

     */
    public boolean isSubsequence(String s, String t) {
        int si=0;
        int ti=0;
        int se=s.length()-1;

        if(s.equals("")) return true;

        while(si<s.length() && ti<t.length()){
            if(s.charAt(si)==t.charAt(ti)){
                if(si==se){ return true; }
                si++; ti++;
            }
            else {
                ti++;
            }
        }

        return false;
    }
}
```
