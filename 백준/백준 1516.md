# 게임 개발

## 문제 분석

- 건물의 번호 : 1~N
- 1 ≤ N ≤ 500
- 각 건물 짓는 데 걸리는 시간 : 10,000 이하 자연수
- 여러 건물을 동시에 지을 수 있음
- 건물을 찾는 데 필요한 최소 시간 구하기

## 풀이

- arr[i]->j : i번째 건물 지은 뒤 j 지을 수 있음
- build : i번째 건물 하나만 짓는 데 걸리는 시간
- indegree[i] : i번째 건물 짓기 전에 지어야 하는 건물의 수

```java
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;

public class P1516 {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        ArrayList<Integer>[] arr = new ArrayList[N + 1];
        int[] build = new int[N + 1];
        int[] indegree = new int[N + 1];

        for (int i = 1; i <= N; i++) {
            arr[i] = new ArrayList<>();
        }

        for (int i = 1; i <= N; i++) {
            build[i] = sc.nextInt();
            int a = sc.nextInt();

            while (a != -1) {
                arr[a].add(i);
                indegree[i]++;
                a = sc.nextInt();
            }
        }

        solve(arr, build, indegree);
    }

    // arr[i]->j : i번째 건물 지은 뒤 j 지을 수 있음
    // build : i번째 건물 하나만 짓는 데 걸리는 시간
    // indegree[i] : 진입차수
    private static void solve(ArrayList<Integer>[] arr, int[] build, int[] indegree) {

        Queue<Integer> queue = new LinkedList<>();
        int[] ans = new int[build.length];

        for (int i = 1; i < indegree.length; i++) {
            if (indegree[i] == 0) {
                queue.add(i);
                ans[i] = build[i];
            }
        }

        while (!queue.isEmpty()) {
            int p = queue.poll();

            for (int a : arr[p]) {
                ans[a] = Math.max(ans[a], build[a] + ans[p]);
                indegree[a]--;
                if (indegree[a] == 0) {
                    queue.add(a);
                }
            }
        }

        for (int i = 1; i < ans.length; i++) {
            System.out.println(ans[i]);
        }

    }
}
```
