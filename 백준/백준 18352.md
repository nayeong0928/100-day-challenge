
결과: 메모리 초과  
A-B가 연결되어 있으면 map을 그릴 때 A->B, B->A 둘 다 체크했는데 도시의 개수가 도로의 개수보다 훨씬 많기 때문인 것으로 보인다 (2 ≤ N ≤ 300,000, 1 ≤ M ≤ 1,000,000)


```java

import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;

public class P18352 {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        int M = sc.nextInt();
        int K = sc.nextInt();
        int X = sc.nextInt();
        int[][] map = new int[N + 1][N + 1];

        for (int i = 0; i < M; i++) {
            int A = sc.nextInt();
            int B = sc.nextInt();

            map[A][B] = 1;
            map[B][A] = 1;
        }

        bfs(map, K, X);
    }

    private static void bfs(int[][] map, int K, int X) {
        boolean[] visited = new boolean[map.length];
        Queue<City> queue = new LinkedList<>();

        visited[X] = true;

        for (int i = 1; i < map.length; i++) {
            if (!visited[i] && map[X][i] == 1) {
                queue.add(new City(i, 1));
                visited[i] = true;
            }
        }

        ArrayList<String> list = new ArrayList<>();

        while (!queue.isEmpty()) {
            City now = queue.poll();

            if (now.dist == K) {
                list.add(Integer.toString(now.num));
            }

            for (int i = 1; i < map.length; i++) {
                if (!visited[i] && map[now.num][i] == 1) {
                    queue.add(new City(i, now.dist + 1));
                    visited[i] = true;
                }
            }
        }

        if (list.isEmpty()) {
            System.out.println("-1");
        } else {
            for (String li : list) {
                System.out.println(li);
            }
        }

    }

    static class City {
        int num;
        int dist;

        public City(int n, int d) {
            num = n;
            dist = d;
        }
    }

}
```
