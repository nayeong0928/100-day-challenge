# 집합의 표현

## 문제 분석

- n: 집합 최대값
- m: 연산 갯수
- 0으로 시작: 합집합
- 1로 시작: 같은 집합인지 확인해서 같으면 yes, 다르면 no 출력

## 풀이

```java

import java.util.Scanner;

public class P1717 {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int m = sc.nextInt();

        int[] grp = new int[n + 1];

        for (int i = 0; i < grp.length; i++) {
            grp[i] = i;
        }

        for (int i = 0; i < m; i++) {
            int c = sc.nextInt();
            int a = sc.nextInt();
            int b = sc.nextInt();

            if (c == 0) {
                union(a, b, grp);
            } else if (c == 1) {
                if (parent(a, grp) == parent(b, grp)) {
                    System.out.println("yes");
                } else {
                    System.out.println("no");
                }
            }

        }
    }

    private static void union(int a, int b, int[] grp) {
        if (a == b) {
            return;
        }
        int A = parent(a, grp);
        int B = parent(b, grp);

        int min = Math.min(A, B);
        int max = Math.max(A, B);

        grp[max] = min;
        grp[a] = min;
        grp[b] = min;
    }

    private static int parent(int a, int[] grp) {
        if (grp[a] == a) {
            return a;
        }

        grp[a] = parent(grp[a], grp);
        return grp[a];
    }
}


```
