# 화학식량

## 문제 분석

- 화학식에 포함되어 있는 모든 원자들의 질량의 합 구하기
- 원자 질량
  - H : 1
  - C : 12
  - O : 16
- 괄호 뒤에 있는 숫자 : 괄호 원소의 개수
- 알파벳 뒤에 있는 숫자 : 바로 앞 원소(한글자)의 개수

## 풀이

- '(' : stack LEFT를 넣음
- ')' : LEFT가 나올 때까지 pop()해서 연산
- H, C, O : 1, 12, 16 stack에 push()
- 숫자 : stack.pop() 해서 숫자에 곱한 뒤 stack에 넣어줌

```java

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Stack;

public class P2257 {

    static final int LEFT = -1;

    public static void main(String[] args) throws IOException {
        // sol("(H)2(O)");
        // System.out.println();
        // sol("CH(CO2H)3");
        // System.out.println();
        // sol("((CH)2(OH2H)(C(H))O)3");

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String s = br.readLine();
        sol(s);
    }

    private static void sol(String str) {

        Stack<Integer> stack = new Stack<>();

        for (int i = 0; i < str.length(); i++) {
            if (str.charAt(i) == '(') {
                stack.push(LEFT);
            } else if (str.charAt(i) == ')') {
                int sum = 0;
                int p = stack.pop();
                while (p != LEFT) {
                    sum += p;
                    p = stack.pop();
                }
                stack.push(sum);
            } else if (str.charAt(i) == 'H') {
                stack.push(1);
            } else if (str.charAt(i) == 'C') {
                stack.push(12);
            } else if (str.charAt(i) == 'O') {
                stack.push(16);
            } else {
                int n = str.charAt(i) - '0';
                int p = stack.pop();
                stack.push(n * p);
            }
        }
        int ans = 0;
        while (!stack.isEmpty()) {
            ans += stack.pop();
        }
        System.out.println(ans);
    }

}
```
