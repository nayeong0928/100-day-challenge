# 트리의 지름 (푸는중)

```java

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.List;
import java.util.Queue;
import java.util.StringTokenizer;

public class sol1167 {

    static int V;
    static List<Edge>[] arr;
    static boolean[] visited;
    static int ans;

    public static void main(String[] args) throws NumberFormatException, IOException {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        V = Integer.parseInt(br.readLine());
        arr = new ArrayList[V + 1];
        visited = new boolean[V + 1];
        ans = -1;

        for (int i = 1; i <= V; i++) {
            arr[i] = new ArrayList<>();
        }

        StringTokenizer st;
        Edge max = new Edge(-1, -1);

        for (int i = 0; i < V; i++) {
            String line = br.readLine();
            st = new StringTokenizer(line);
            int from = Integer.parseInt(st.nextToken(" "));

            while (true) {
                int to = Integer.parseInt(st.nextToken(" "));
                if (to == -1)
                    break;
                int dist = Integer.parseInt(st.nextToken(" "));
                Edge edge = new Edge(to, dist);
                arr[from].add(edge);
                if (max.dist < edge.dist) {
                    max = edge;
                }
            }
        }

        Queue<Edge> queue = new LinkedList<>();
        queue.add(max);
        visited[max.node] = true;

        while (!queue.isEmpty()) {
            Edge edge = queue.poll();
            ans = Math.max(edge.dist, ans);

            for (Edge e : arr[edge.node]) {
                if (!visited[edge.node]) {
                    visited[edge.node] = true;
                    queue.add(new Edge(e.node, edge.dist + e.dist));
                }
            }
        }

        System.out.println(ans);
    }

    static class Edge {
        int node;
        int dist;

        public Edge(int n, int d) {
            this.node = n;
            this.dist = d;
        }
    }
}
```
